AND MMT.TRANSACTION_TYPE_ID = MTT.TRANSACTION_TYPE_ID

AND MSI.INVENTORY_ITEM_STATUS_CODE = 'Active'
AND MSI.INVENTORY_ITEM_ID = '132085'

--AND :P_END_DATE

select MCBK.SEGMENT4, FFVL.DESCRIPTION, MCBK.SEGMENT5, FFVL1.DESCRIPTION
  from MTL_SYSTEM_ITEMS_KFV msik,
       MTL_ITEM_CATEGORIES MIC,
       MTL_CATEGORIES_B_KFV MCBK,
       FND_FLEX_VALUES_VL FFVL,
       FND_FLEX_VALUES_VL FFVL1
 where msik.INVENTORY_ITEM_ID = 179930
   and msik.INVENTORY_ITEM_ID = mic.INVENTORY_ITEM_ID
   and msik.ORGANIZATION_ID = mic.ORGANIZATION_ID
   and mic.CATEGORY_SET_ID = 1100000121
   AND MIC.CATEGORY_ID = MCBK.CATEGORY_ID
   AND MCBK.SEGMENT4 = FFVL.FLEX_VALUE
   AND MCBK.SEGMENT5 = FFVL1.FLEX_VALUE
   AND FFVL1.FLEX_VALUE_SET_ID = 1019038
   AND FFVL.FLEX_VALUE_SET_ID = 1019037
